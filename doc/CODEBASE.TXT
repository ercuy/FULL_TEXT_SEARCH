068   def find_missing_pdfs(root: str):
069       """Doorzoekt rootmap en retourneert lijst dicts met bronbestand en verwacht PDF-pad."""
070       root = Path(root)
071       results = []

077               continue
078   
079           if path.suffix.lower() in [".doc", ".docx"]:
080               expected_pdf = path.with_suffix(".pdf")
081               if not expected_pdf.exists():
082                   results.append({

084                       "expected_pdf": str(expected_pdf),

091       """Schrijft JSON-lijst naar bestand."""

099       lines = [

104   
105       for item in data:
106           src = item["source"].replace("\\", "/")
107           pdf = item["expected_pdf"].replace("\\", "/")
108           link = f'<a href="file:///{src}">{Path(src).name}</a>'
109           lines.append(f"<li>{link} â†’ verwacht: {pdf}</li>")

004   from typing import Optional
005   import re
006   
007   import pypdf
008   
009   
010   def extract_text_basic(path: str, max_chars: Optional[int] = None) -> str:
011       """
012       Eenvoudige tekstextractie uit PDF.
013       Audit-proof: geen stille fouten, altijd string terug.
014       """
015       try:
016           reader = pypdf.PdfReader(path)
017           texts = []
018           for page in reader.pages:
019               t = page.extract_text() or ""
020               texts.append(t)
021               if max_chars is not None and sum(len(x) for x in texts) >= max_chars:
022                   break
023           result = "\n".join(texts)
024           if max_chars is not None:
025               return result[:max_chars]
026           return result
027       except Exception as e:
028           print(f"[EXTRACT][ERROR] {path}: {e}")
029           return ""
030   
031   
032   def has_text_objects(path: str) -> bool:
033       """
034       Heuristiek: controleer of er tekstobjecten/fonts aanwezig zijn.
035       Voor eenvoud: als extract_text_basic > 0, beschouwen we als tekst-PDF.
036       Voor meer diepgang kan PyMuPDF gebruikt worden, maar hier houden we het compact.
037       """
038       text = extract_text_basic(path, max_chars=500)
039       return len(text.strip()) > 0
040   
041   
042   def detect_language(text: str) -> str:
043       """
044       Zeer eenvoudige taalheuristiek NL/EN.
045       Voor echte detectie kan langdetect gebruikt worden, maar we houden het licht.
046       """
047       nl_markers = ["de ", "het ", "een ", "rapport", "kalibratie", "kalibreren", "monster"]
048       en_markers = ["the ", "report", "calibration", "sample", "quality"]
049   
050       nl_score = sum(text.lower().count(m) for m in nl_markers)
051       en_score = sum(text.lower().count(m) for m in en_markers)
052   
053       if nl_score >= en_score:
054           return "nl"
055       return "en"
056   
057   
058   def extract_title(text: str) -> str:
059       """
060       Neem de eerste niet-lege regel als titel.
061       """
062       for line in text.splitlines():
063           line = line.strip()
064           if line:
065               return line[:200]
066       return ""
067   
068   
069   
070   def extract_search_text(text: str, max_words: int = 400) -> str:
071       """
072       Genereer zoekbare contexttekst:
073       - geen cijfers
074       - voldoende letters
075       - beperkt aantal woorden
076       """
077       words = []
078       for token in re.findall(r"[A-Za-zÀ-ÿ]{3,}", text):
079           words.append(token.lower())
080           if len(words) >= max_words:
081               break
082       return " ".join(words)

056   def _connect_db(db_path: str) -> sqlite3.Connection:
057       os.makedirs(os.path.dirname(db_path), exist_ok=True)
058       conn = sqlite3.connect(db_path)
059       conn.execute("PRAGMA journal_mode=WAL;")

064   def _ensure_column(conn: sqlite3.Connection, table: str, column: str, ddl: str) -> None:
065       """
066       Zorgt dat een kolom bestaat (migratie-proof).
067       - We checken PRAGMA table_info.
068       - Als de kolom ontbreekt, voeren we 1x ALTER TABLE uit.
069       """
070       cols = [r[1] for r in conn.execute(f"PRAGMA table_info({table})").fetchall()]
071       if column not in cols:
072           conn.execute(ddl)
073           conn.commit()

076   def _make_search_text(raw_text: str, max_words: int = 400) -> str:
077       """
078       Maak een zoek-anker tekst (deterministisch):
079       - enkel woorden met letters (geen cijfers)
080       - minimum 3 letters
081       - lowercased
082       - capped op max_words
083       Waarom: betere recall/precision dan 'summary' voor FTS.
084       """
085       tokens = re.findall(r"[A-Za-zÀ-ÿ]{3,}", raw_text)
086       tokens = [t.lower() for t in tokens][:max_words]
087       return " ".join(tokens)

152   # ---------------------------------------------------------------------
153   # Update FTS (portable)
154   # ---------------------------------------------------------------------
155   def _update_fts(conn: sqlite3.Connection, record: Dict[str, Any]) -> None:
156       """
157       Update FTS5 row voor dit document.
158       We gebruiken hier bewust:
159       - DELETE FROM files_fts WHERE rowid=?
160       - INSERT INTO files_fts(...)
161       Dit is het meest portable en voorkomt "SQL logic error" varianten.
162       """
163       row = conn.execute("SELECT id FROM files WHERE path = ?", (record["path"],)).fetchone()
164       if not row:
165           return
166       doc_id = row[0]
167   
168       conn.execute("DELETE FROM files_fts WHERE rowid = ?", (doc_id,))
169       conn.execute(
170           """
171           INSERT INTO files_fts(rowid, content, tags, summary, path)
172           VALUES (?, ?, ?, ?, ?)
173           """,
174           (
175               doc_id,
176               record["search_text"],  # <-- hier zit de 'paar honderd woorden zonder cijfers'
177               record["tags"],
178               record["summary"],
179               record["path"],
180           ),
181       )
182       conn.commit()

187   # ---------------------------------------------------------------------
188   def index_all() -> None:
189       settings = load_settings()
190       init_db()
191       conn = _connect_db(settings.paths.index_db)
192   
193       batch: List[Dict[str, Any]] = []
194       batch_size = settings.indexing.batch_size
195   
196       for meta in scan_and_classify():
197           path = meta["path"]
198           print(f"[INDEX] Processing {path}")
199   
200           text = extract_text_basic(path)
201           if len(text.strip()) < settings.file_handling.min_text_length:
202               print(f"[INDEX][SKIP] Too little text in {path}")
203               continue
204   
205           title = extract_title(text)
206           tags = extract_tags(text)
207           patterns = extract_patterns(text)
208           summary = summarize_with_language(text, max_sentences=settings.summaries.max_sentences)
209   
210           search_text = _make_search_text(text, max_words=400)
211   
212           record = {
213               "path": path,
214               "size": meta["size"],
215               "mtime": meta["mtime"],
216               "ctime": meta["ctime"],
217               "title": title,
218               "tags": ",".join(tags),
219               "project_codes": ",".join(patterns.get("project_code", [])),
220               "sample_ids": ",".join(patterns.get("sample_id", [])),
221               "dates": ",".join(patterns.get("date", [])),
222               "summary": summary,
223               "search_text": search_text,
224           }
225   
226           batch.append(record)
227   
228           if len(batch) >= batch_size:
229               _flush_batch(conn, batch, settings)
230               batch.clear()
231   
232       if batch:
233           _flush_batch(conn, batch, settings)
234   
235       conn.close()
236       print("[INDEX] Completed indexing")
237

239   def _flush_batch(conn: sqlite3.Connection, batch: List[Dict[str, Any]], settings) -> None:
240       for rec in batch:
241           try:
242               _upsert_file(conn, rec)
243               if settings.indexing.fts5_enabled:
244                   _update_fts(conn, rec)
245           except Exception as e:
246               print(f"[INDEX][ERROR] {rec['path']}: {e}")

250   # Entry point
251   # ---------------------------------------------------------------------
252   if __name__ == "__main__":
253       index_all()

045   def is_scanned_pdf(path: str, settings) -> bool:
046       """
047       Detecteer gescande PDF's op basis van:
048       - tekstlengte
049       - aanwezigheid van tekstobjecten/fonts
050       """
051       text = extract_text_basic(path, max_chars=2000)
052       if settings.file_handling.detect_scanned_pdf.get("check_text_length", True):
053           if len(text.strip()) < settings.file_handling.min_text_length:

087                   "mtime": stat.st_mtime,
088                   "ctime": stat.st_ctime,

010       """
011       Zeer eenvoudige zinssegmentatie.
012       Genoeg voor technische rapporten.
013       """
014       raw = text.replace("\n", " ")
015       parts = [p.strip() for p in raw.split(".") if p.strip()]
016       return [p + "." for p in parts]

019   def summarize_text(text: str, max_sentences: int = 3) -> str:
020       """
021       Extractieve samenvatting: neem de eerste zinnen die inhoudelijk lijken.
022       Geen AI, volledig deterministisch.
023       """
024       if not text.strip():
025           return ""
026   
027       sentences = _split_sentences(text)
028       if not sentences:
029           return ""
030   
031       # Eenvoudige heuristiek: neem de eerste zinnen die langer zijn dan X
032       filtered = [s for s in sentences if len(s) > 40]
033       if not filtered:
034           filtered = sentences
035   
036       summary = " ".join(filtered[:max_sentences])
037       return summary

017       """
018       settings = load_settings()
019       vocab = [v.lower() for v in settings.tags.vocabulary]

031       for term in vocab:
032           if term in norm:
033               # Gebruik originele vorm uit vocabularium
034               for original in settings.tags.vocabulary:
035                   if original.lower() == term:
036                       found.add(original)
037                       break
038   
039       # Beperk aantal tags
040       tags = sorted(found)
041       return tags[: settings.tags.max_tags_per_document]

020       log_dir = Path(settings.paths.log_folder)

022       with open(log_dir / "ui_server.log", "a", encoding="utf-8") as f:

026   def _is_allowed_path(path: str, roots: List[str]) -> bool:
027       try:
028           p = os.path.normcase(os.path.abspath(path))
029           for root in roots:
030               r = os.path.normcase(os.path.abspath(root))
031               if os.path.commonpath([p, r]) == r:
032                   return True
033       except Exception:
034           pass
035       return False

068       # ------------------------------------------------------------
069       # PDF OPEN (SERVER-SIDE, ROBUST)
070       # ------------------------------------------------------------
071       @app.route("/pdf/<int:doc_id>", methods=["GET"])
072       def open_pdf(doc_id: int):
073           doc = get_document(doc_id)
074           if not doc or not doc.get("path"):
075               return "Document not found", 404
076   
077           path = doc["path"]
078           if not _is_allowed_path(path, settings.paths.root_locations):
079               return "Access denied", 403
080   
081           _log(settings, f"OPEN_PDF id={doc_id} path={path}")
082           return send_file(path, as_attachment=False)
083   
084       # ------------------------------------------------------------
085       # ZIP DOWNLOAD (MAX 25, WHITELIST, LOGGED)
086       # ------------------------------------------------------------
087       @app.route("/zip", methods=["POST"])
088       def zip_download():
089           data = request.get_json(silent=True) or {}
090           files = data.get("files") or []
091   
092           if not files:
093               return jsonify({"error": "Geen bestanden opgegeven"}), 400
094           if len(files) > 25:
095               return jsonify({"error": "Max 25 bestanden toegestaan"}), 400
096

098           added, skipped = 0, 0
099   
100           with zipfile.ZipFile(mem, "w", zipfile.ZIP_DEFLATED) as z:
101               base_root = settings.paths.root_locations[0]
102               for f in files:
103                   if not _is_allowed_path(f, settings.paths.root_locations):
104                       skipped += 1
105                       continue
106                   if not os.path.exists(f):
107                       skipped += 1
108                       continue
109   
110                   rel = os.path.relpath(f, start=base_root)
111                   z.write(f, arcname=rel)
112                   added += 1
113   
114           mem.seek(0)
115           _log(settings, f"ZIP files={len(files)} added={added} skipped={skipped}")
116           return send_file(mem, mimetype="application/zip",
117                            as_attachment=True,
118                            download_name="documents.zip")
119   
120       # ------------------------------------------------------------
121       # METADATA JSON (OPTIONEEL, KLAAR VOOR CLIENT-SIDE)
122       # ------------------------------------------------------------
123       @app.route("/metadata", methods=["GET"])
124       def metadata():
125           import sqlite3
126           conn = sqlite3.connect(settings.paths.index_db)
127           conn.row_factory = sqlite3.Row
128           rows = conn.execute("""
129               SELECT id, path, title, tags, summary,
130                      project_codes, sample_ids, dates, mtime, ctime
131               FROM files
132               ORDER BY mtime DESC
133               LIMIT 5000
134           """).fetchall()
135           conn.close()
136           return jsonify([dict(r) for r in rows])
137   
138       return app
